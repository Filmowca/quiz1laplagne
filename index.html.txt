function addPlayer() {
  const inp  = document.getElementById('name-inp');
  const name = inp.value.trim();
  if (!name) return;
  if (players.length >= 15) { alert('Maksymalnie 15 graczy!'); return; }
  const dup = players.some(p => p.name.toLowerCase() === name.toLowerCase());
  if (dup) { inp.value = ''; inp.focus(); return; }
  players.push({ name: name, color: COLORS[players.length % COLORS.length], score: [0,0,0] });
  inp.value = '';
  inp.focus();
  refreshSetup();
}

function removePlayer(i) {
  players.splice(i, 1);
  players.forEach((p, idx) => p.color = COLORS[idx % COLORS.length]);
  refreshSetup();
}

      '<button class="del" onclick="removePlayer(' + i + ')">✕</button>';
    list.appendChild(chip);
  });
  const n = players.length;
  count.textContent = n + (n === 1 ? ' gracz' : ' graczy');
  title.textContent = n > 0 ? 'LISTA GRACZY (' + n + ')' : 'LISTA GRACZY';
  btn.disabled = n < 2;
}

// ════════════════════════════════════════
// BEGIN QUIZ
// ════════════════════════════════════════
function beginQuiz() {
  if (players.length < 2) return;
  players.forEach(p => p.score = [0,0,0]);
  cRound = 0; cQ = 0; cPlayer = 0; qResults = [];
  goRoundIntro(0);
}

function goRoundIntro(idx) {
  cRound  = idx;
  cQ      = 0;
  cPlayer = 0;
  qResults = [];
  document.getElementById('intro-num').textContent   = idx + 1;
  document.getElementById('intro-title').textContent = ROUNDS[idx].title;
  document.getElementById('intro-desc').textContent  =
    'Każdy z ' + players.length + ' graczy odpowiada na każde pytanie';
  showScreen('s-intro');
}

function startQuiz() {
  showScreen('s-quiz');
  renderQ();
}

// ════════════════════════════════════════
// SHUFFLE
// ════════════════════════════════════════
function doShuffle(qd) {
  const correct = qd.a[qd.c];
  const arr = qd.a.slice().sort(() => Math.random() - 0.5);
  return { answers: arr, correctIdx: arr.indexOf(correct) };
}

// ════════════════════════════════════════
// RENDER QUESTION
// ════════════════════════════════════════
function renderQ() {
  const round = ROUNDS[cRound];
  const qd    = round.questions[cQ];
  const total = round.questions.length;

  if (cPlayer === 0) {
    shuffled = doShuffle(qd);
    qResults[cQ] = players.map(() => null);
  }

  // un-blur
  document.getElementById('q-panel').classList.remove('blurred');
  document.getElementById('a-panel').classList.remove('blurred');

  // topbar
  document.getElementById('pr-left').textContent  = 'Pytanie ' + (cQ+1) + '/' + total;
  document.getElementById('pr-right').textContent = 'TURA ' + (cRound+1);
  document.getElementById('pr-fill').style.width  = (cQ / total * 100) + '%';

  const p = players[cPlayer];
  document.getElementById('turn-dot').style.background = p.color;
  document.getElementById('turn-name').textContent     = p.name;

  // question
  document.getElementById('q-tag').textContent  = 'PYTANIE ' + (cQ+1);
  document.getElementById('q-text').textContent = qd.q;

  // answers
  const panel = document.getElementById('a-panel');
  panel.innerHTML = '';
  const LBLS = ['A','B','C','D'];
  shuffled.answers.forEach(function(ans, i) {
    const btn = document.createElement('button');
    btn.className = 'ans-btn';
    btn.innerHTML = '<span class="ans-lbl">' + LBLS[i] + '</span>' + ans;
    btn.addEventListener('click', function() { pickAnswer(i, btn); });
    panel.appendChild(btn);
  });

  // feedback
  const fb = document.getElementById('feedback');
  fb.className = '';
  fb.textContent = '';

  // player strip
  renderStrip();

  // start timer
  startTimer();
}

// ════════════════════════════════════════
// TIMER
// ════════════════════════════════════════
function startTimer() {
  stopTimer();
  timerSecs = TIMER_TOTAL;
  updateTimerUI();
  timerInt = setInterval(function() {
    timerSecs--;
    updateTimerUI();
    if (timerSecs <= 0) {
      stopTimer();
      onTimeUp();
    }
  }, 1000);
}

function stopTimer() {
  if (timerInt) { clearInterval(timerInt); timerInt = null; }
}

function updateTimerUI() {
  document.getElementById('timer-num').textContent = timerSecs;
  const frac   = timerSecs / TIMER_TOTAL;
  const offset = CIRC * (1 - frac);
  const circle = document.getElementById('timer-circle');
  circle.style.strokeDashoffset = offset;
  // color: green → yellow → red
  circle.style.stroke = frac > 0.5 ? '#2ecc71' : frac > 0.25 ? '#f4c430' : '#e74c3c';
}

function onTimeUp() {
  // blur only the question — answers remain clickable so host can still pick
  document.getElementById('q-panel').classList.add('blurred');
  // mark current player as err (no answer) — can still be overridden if host taps
  qResults[cQ][cPlayer] = 'err';
  renderStrip();
  // advance after 3s (gives host time to tap an answer if needed)
  setTimeout(advance, 3000);
}

// ════════════════════════════════════════
// PICK ANSWER
// ════════════════════════════════════════
function pickAnswer(idx, btn) {
  stopTimer();
  document.querySelectorAll('.ans-btn').forEach(b => b.disabled = true);

  const p = players[cPlayer];
  const isCorrect    = (idx === shuffled.correctIdx);
  const isLastPlayer = (cPlayer === players.length - 1);

  qResults[cQ][cPlayer] = isCorrect ? 'ok' : 'err';

  if (isCorrect) {
    p.score[cRound]++;
    const tot = p.score.reduce((a,b) => a+b, 0);
    const fb  = document.getElementById('feedback');
    fb.className = 'show fb-ok';
    fb.textContent = isLastPlayer
      ? '✓ ' + p.name + ' – punkt! (' + tot + ' łącznie)  →  Poprawna odpowiedź podświetlona!'
      : '✓ ' + p.name + ' – punkt! Łącznie: ' + tot + ' pkt';
  } else {
    const fb = document.getElementById('feedback');
    fb.className = 'show fb-err';
    fb.textContent = isLastPlayer
      ? '✗ ' + p.name + ' – błąd!  →  Poprawna odpowiedź podświetlona!'
      : '✗ ' + p.name + ' – błąd!';
  }

  // Reveal correct answer only after LAST player
  if (isLastPlayer) {
    const btns = document.querySelectorAll('.ans-btn');
    btns.forEach(function(b, i) {
      if (i === shuffled.correctIdx) b.classList.add('ans-correct');
      else b.classList.add('ans-wrong');
    });
  }

  renderStrip();
  const delay = isLastPlayer ? 2200 : 1100;
  setTimeout(advance, delay);
}

// ════════════════════════════════════════
// ADVANCE
// ════════════════════════════════════════
function advance() {
  cPlayer++;
  if (cPlayer < players.length) {
    renderQ();
  } else {
    cPlayer = 0;
    cQ++;
    if (cQ < ROUNDS[cRound].questions.length) {
      renderQ();
    } else {
      qResults = [];
      showLeaderboard();
    }
  }
}

// ════════════════════════════════════════
// PLAYER STRIP
// ════════════════════════════════════════
function renderStrip() {
  const strip = document.getElementById('player-strip');
  strip.innerHTML = '';
  players.forEach(function(p, i) {
    const res = (qResults[cQ] && qResults[cQ][i] !== undefined) ? qResults[cQ][i] : null;
    const card = document.createElement('div');

    let cls = 'pcard';
    if      (i === cPlayer) cls += ' pc-active';
    else if (res === 'ok')  cls += ' pc-ok pc-done';
    else if (res === 'err') cls += ' pc-err pc-done';

    const tick = res === 'ok' ? '✓' : res === 'err' ? '✗' : '';
    const tickHtml = tick ? '<div class="pc-tick">' + tick + '</div>' : '';
    const total = p.score.reduce((a,b) => a+b, 0);

    card.className = cls;
    card.id = 'pc-' + i;
    card.innerHTML =
      '<div class="pc-av" style="background:' + p.color + '">' + p.name[0].toUpperCase() + tickHtml + '</div>' +
      '<div class="pc-name">' + p.name + '</div>' +
      '<div class="pc-score">' + total + '</div>';
    strip.appendChild(card);
  });
  const active = document.getElementById('pc-' + cPlayer);
  if (active) active.scrollIntoView({ inline:'center', block:'nearest' });
}

// ════════════════════════════════════════
// LEADERBOARD
// ════════════════════════════════════════
function showLeaderboard() {
  stopTimer();
  const r = cRound;
  document.getElementById('board-round-lbl').textContent = 'PO TURZE ' + (r+1);

  const sorted = players.slice().sort(function(a,b) {
    return b.score.reduce((x,y) => x+y, 0) - a.score.reduce((x,y) => x+y, 0);
  });

  const bestRound = players.slice().sort((a,b) => b.score[r] - a.score[r])[0];
  document.getElementById('bb-name').textContent = bestRound.name;
  document.getElementById('bb-pts').textContent  = bestRound.score[r] + ' pkt w tej turze';

  const maxTotal = ROUNDS.reduce((a, rd) => a + rd.questions.length, 0);
  const right = document.getElementById('board-right');
  right.innerHTML = '';

  const PC = ['b1','b2','b3'];
  const CC = ['gold','silver','bronze'];
  sorted.forEach(function(p, pos) {
    const total = p.score.reduce((a,b) => a+b, 0);
    const row = document.createElement('div');
    row.className = 'brow ' + (PC[pos] || '');
    row.innerHTML =
      '<div class="bp ' + (CC[pos]||'') + '">' + (pos+1) + '</div>' +
      '<div class="bav" style="background:' + p.color + '">' + p.name[0].toUpperCase() + '</div>' +
      '<div class="bnm">' + p.name + '</div>' +
      '<div class="brp">+' + p.score[r] + ' pkt</div>' +
      '<div class="btt">' + total + '</div>' +
      '<div class="bbar-w"><div class="bbar" style="width:' + Math.round(total/maxTotal*100) + '%"></div></div>';
    right.appendChild(row);
  });

  const btn = document.getElementById('btn-board-next');
  if (r < ROUNDS.length - 1) {
    btn.textContent = 'Tura ' + (r+2) + ' →';
    btn.onclick     = function() { goRoundIntro(r+1); };
  } else {
    btn.textContent = '🏆 Wynik końcowy';
    btn.onclick     = showFinal;
  }
  showScreen('s-board');
}

// ════════════════════════════════════════
// FINAL
// ════════════════════════════════════════
function showFinal() {
  const maxTotal = ROUNDS.reduce((a, r) => a + r.questions.length, 0);
  const sorted   = players.slice().sort(function(a,b) {
    return b.score.reduce((x,y) => x+y, 0) - a.score.reduce((x,y) => x+y, 0);
  });
  const medals = ['🥇','🥈','🥉'];
  const list   = document.getElementById('final-list');
  list.innerHTML = '';
  sorted.forEach(function(p, pos) {
    const total = p.score.reduce((a,b) => a+b, 0);
    const pct   = Math.round(total / maxTotal * 100);
    const row   = document.createElement('div');
    row.className = 'frow ' + (['f1','f2','f3'][pos] || '');
    row.innerHTML =
      '<div class="fm">' + (medals[pos] || (pos+1)+'.') + '</div>' +
      '<div class="fav" style="background:' + p.color + '">' + p.name[0].toUpperCase() + '</div>' +
      '<div class="fnm">' + p.name + '</div>' +
      '<div class="fpt">' + total + '<span class="fpc"> ' + pct + '%</span></div>';
    list.appendChild(row);
  });
  showScreen('s-final');
  launchConfetti();
}

function launchConfetti() {

  const cols = ['#f4c430','#00b4d8','#2ecc71','#e74c3c','#fff','#c5e4f5'];

      '--x:' + x + 'vw;' +

      '--dur:' + (Math.random()*2+2).toFixed(1) + 's;' +
      '--dl:'  + (Math.random()*1.4).toFixed(2) + 's;' +

      'background:' + cols[i % cols.length] + ';' +

  }

}

// FULLSCREEN
// ════════════════════════════════════════
function toggleFS() {
  const btn = document.getElementById('btn-fs');
  const el  = document.documentElement;
  const inFS = document.fullscreenElement || document.webkitFullscreenElement;
  if (!inFS) {
    const req = el.requestFullscreen || el.webkitRequestFullscreen || el.mozRequestFullScreen || el.msRequestFullscreen;
    if (req) {
      req.call(el);
      btn.textContent = '✕';
      btn.title = 'Wyjdź z pełnego ekranu';
    } else {
      // iOS Safari fallback — hide status bar via scroll trick
      window.scrollTo(0, 1);
      btn.style.display = 'none';
    }
  } else {
    const exit = document.exitFullscreen || document.webkitExitFullscreen || document.mozCancelFullScreen;
    if (exit) exit.call(document);
    btn.textContent = '⛶';
    btn.title = 'Pełny ekran';
  }
}

// Auto-hide fullscreen button after 4s of no interaction
let fsHideTimer;
function resetFsTimer() {
  const btn = document.getElementById('btn-fs');
  btn.style.opacity = '1';
  clearTimeout(fsHideTimer);
  fsHideTimer = setTimeout(() => { btn.style.opacity = '0.15'; }, 4000);
}
document.addEventListener('touchstart', resetFsTimer);
document.addEventListener('click', resetFsTimer);
resetFsTimer();
